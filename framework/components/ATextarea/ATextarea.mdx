# Textareas

## Install

Integrate with your build to [auto-import](/#integrating), or add an import in your component:

```
import {ATextarea} from "@cisco-sbg-ui/magna-react";
```

## Usage

<Playground
  code={`() => {
  const [autoGrow, setAutoGrow] = useState(true);
  const [value, setValue] = useState(LoremIpsum);
  return (
    <div style={{maxWidth: 250}}>
      <div className="mb-4">
        <ACheckbox checked={autoGrow} onClick={() => setAutoGrow(!autoGrow)}>
          Toggle autoGrow
        </ACheckbox>
        <ATextarea
          label="Label"
          placeholder="Normal with autoGrow"
          autoGrow={autoGrow}
          disableGrammarly
          value={value}
          onChange={(e) => {
            setValue(e.target.value);
          }}
          hint="Hint: Normal"
        />
      </div>
    </div>
  );
}
`}
/>

## Validation

<Playground
  code={`() => {
  const [comments, setComments] = useState("");
  return (
    <div>
      <h3>Required Field</h3>
      <ATextarea
        autoFocus
        label="Comments"
        onChange={(e) => {
          setComments(e.target.value);
        }}
        value={comments}
        required
        rules={[
          {
            test: (v) =>
              v.length < 10 || "Comments must be less than 10 characters",
            level: "danger"
          }
        ]}
        validateOnBlur
        hint="Keep it short."
      />
    </div>
  );
}
`}
/>
