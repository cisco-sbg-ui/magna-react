# Stepper

## Install

Integrate with your build to [auto-import](/#integrating), or add an import in your component:

```
import {AStepper, AStep, AStepTitle, AStepDescription} from "@cisco-sbg-ui/magna-react";
```

      description:

    },

      title: "Default",

  return (
    <div>
      <div style={{border: "solid 2px #333"}} className="my-2 pa-2">

      </div>
      <h3>Stepper with icon:</h3>
      <AStepper className="AStepper">

          const stepNumber = index + 1;
          return (
            <AStep
              key={stepNumber}
              stepNumber={stepNumber}
              active={active === stepNumber}
              visited={active > stepNumber}
              disabled={stepNumber === 2}
              className="AStep"
              showIconOnVisited={true}>
              <AStepTitle className="AStepTitle">
                {item.title}
              </AStepTitle>
              <AStepDescription className="AStepDescription">{item.description}</AStepDescription>
              </AStep>
          );
        })}
      </AStepper>
      <AButton onClick={() => active > 1 && setActive(active - 1)}>
        Previous
      </AButton>
      <AButton onClick={() => active <= data.length && setActive(active + 1)}>
        Next
      </AButton>
    </div>
  );
}
`}
