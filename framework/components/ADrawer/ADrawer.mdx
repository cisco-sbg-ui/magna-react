# Drawers

## Install

Integrate with your build to [auto-import](/#integrating), or add an import in your component:

```
import {ADrawer} from "@cisco-sbg-ui/magna-react";
```

## Usage

<Playground
  code={`() => {
    const [isLeftOpen, setIsLeftOpen] = useState(false);
    const [isRightOpen, setIsRightOpen] = useState(false);
    const [isBottomOpen, setIsBottomOpen] = useState(false);
    
    return (
      <>
        <p>Left slide-in</p>
        <AButton onClick={() => setIsLeftOpen(p => !p)}>Toggle Left</AButton>
        <ADrawer isOpen={isLeftOpen}>
          <ADrawerContent>
            Left Drawer <button onClick={() => setIsLeftOpen(false)}>x</button>
          </ADrawerContent>
        </ADrawer>
        
        <p>Right slide-in</p>
        <AButton onClick={() => setIsRightOpen(p => !p)}>Toggle Right</AButton>
        <ADrawer slideIn='right' isOpen={isRightOpen}>
          <ADrawerContent>
            Right Drawer <button onClick={() => setIsRightOpen(false)}>x</button>
          </ADrawerContent>
        </ADrawer>
        
        <p>Bottom slide-in</p>
        <AButton onClick={() => setIsBottomOpen(p => !p)}>Toggle Bottom</AButton>
        <ADrawer slideIn='bottom' isOpen={isBottomOpen}>
          <ADrawerContent>
            Bottom Drawer <button onClick={() => setIsBottomOpen(false)}>x</button>
          </ADrawerContent>    
        </ADrawer>
      </>
    )
  }`}
/>
