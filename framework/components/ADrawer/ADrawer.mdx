### Usage

By default, `ADrawer` renders over the entire page (as a Modal) on the left side of the screen, but you can configure the drawer to your needs with the following customizations:

<Playground
  withTwoPanes

  return (
    <>
      <AButton
        onClick={() => setIsOpen(true)}

        Open
      </AButton>
      <ADrawer
        isOpen={isOpen}
        closeBtnOnClick={() => setIsOpen(false)}>
        <ADrawerContent>

  )
  }`}

#### Slide-in Directions

The drawer can slide into the view from either the left, right, or bottom of the page.

<Playground
  code={`function Example() {
  const initialState = {
    left: false,
    right: false,
    bottom: false,
  };
  const [drawerState, setDrawerState] = useState(initialState);
  useEscapeKeydown({
    isEnabled: Object.values(drawerState).some(isOpen => isOpen),
    onKeydown: () => setDrawerState(initialState)
  });
  return ["Left", "Right", "Bottom"].map((direction) => {
    return (
      <>
        <AButton
          onClick={() =>
            setDrawerState((states) => ({
              ...states,
              [direction]: true,
            }))
          }
          secondary
        >
          {direction}
        </AButton>
        <ADrawer
          slideIn={direction.toLowerCase()}
          isOpen={drawerState[direction]}
          closeBtnOnClick={() =>
            setDrawerState((states) => ({
              ...states,
              [direction]: false,
            }))
          }
        >
          <ADrawerContent>
            <h3 id="left-drawer-title" className="mt-1">
              {direction} Drawer
            </h3>
          </ADrawerContent>
        </ADrawer>
      </>
    );
  });
  }`}
/>

#### Drawer Positioning Strategies

By default, `ADrawer` is rendered as a fixed element. This means that the drawer will pan the entire viewport, covering any content underneath <Footnote num={0} />

To contain the drawer within something other than the viewport, you can render the drawer as an element with `absolute` <Footnote num={1} /> or `relative` positioning with the `position` prop. `absolute` positioning is best useful when you want the drawer to cover the content it pans over, whereas `relative` positioning will push the content aside.

0: Assumes <a target="_blank" href="https://developer.mozilla.org/en-US/docs/Web/CSS/position#values"> one of the drawer's ancestors _does not_ have a `transform`, `perspective`, or `filter` property set to something other than none</a>.

1: Please note that when rendering the drawer with a position of `absolute`, you will also need to ensure it has the proper <a target="_blank" href="https://developer.mozilla.org/en-US/docs/Web/CSS/Containing_block#identifying_the_containing_block">containing block</a> relative to where you'd like the drawer to be positioned. This is no different than how you'd configure any other element with `absolute` positioning.

#### Modal Options

Becuase `ADrawer` is a `fixed` element by default, it also defaults to rendering as [a modal](/components/modal). This can be easily overriden by passing `false` to the `asModal` prop of `ADrawer`.

Please note that when `ADrawer` is rendered as a modal, focus will be trapped inside the opened drawer's content, and scrolling outside of the drawer will be locked.

<Playground
  code={`function Example() {
    const modalContentRef = useRef();
    const [modalOpen, setModalOpen] = useState(false);
    usePopupQuickExit({
      popupRef: modalContentRef,
      isEnabled: modalOpen,
      onExit: () => setModalOpen(false)
    });

    const nonModalContentRef = useRef();
    const [nonModalOpen, setNonModalOpen] = useState(false);
    usePopupQuickExit({
      popupRef: nonModalContentRef,
      isEnabled: nonModalOpen,
      onExit: () => setNonModalOpen(false)
    });

    return (
      <>
        <p>Modal Drawer</p>
        <AButton
            aria-label='Open modal drawer'
            onClick={() => setModalOpen(true)}>
          Open
        </AButton>
        <ADrawer
            aria-label='Drawer that behaves like a modal.'
            ref={modalContentRef}
            asModal={true}
            isOpen={modalOpen}
            slideIn='right'>
            <MockDrawerContent
                title='Modal Drawer'
                content='A Drawer that behaves as a modal.'
                onClose={() => setModalOpen(false)} />
        </ADrawer>

        <p>Non-Modal Drawer</p>
        <AButton
            aria-label='Open non-modal drawer'
            onClick={() => setNonModalOpen(prevState => !prevState)}>
            Toggle
        </AButton>
        <ADrawer
            ref={nonModalContentRef}
            asModal={false}
            isOpen={nonModalOpen}
            slideIn='right'>
          <MockDrawerContent
              title='Non-Modal Drawer'
              content='Focus is not trapped and scrolling is not locked.'
              onClose={() => setNonModalOpen(false)} />
        </ADrawer>
      </>
    )
    function MockDrawerContent({ content, onClose, title }) {
      return (
        <>
          <AListItem twoLine>
            <AListItemContent>
              <AListItemTitle><h4><b>{title}</b></h4></AListItemTitle>
            </AListItemContent>
            <AListItemAction>
              <AButton onClick={onClose} tertiaryAlt icon>
                <AIcon>close</AIcon>
              </AButton>
            </AListItemAction>
          </AListItem>
          <AListItem twoLine>
            <AListItemContent style={{paddingTop: '0'}}>
              {content}
            </AListItemContent>
          </AListItem>
          <div style={{width: '100%', position: 'absolute', bottom: '0', height: '50px'}}>
            <ADivider style={{margin: 0}} />
            <AListItem twoLine>
              <AListItemContent><AButton>Footer Content</AButton></AListItemContent>
            </AListItem>
          </div>
        </>
      )
    }

}`}
/>

## Accessibility Notes

By default, `ADrawer` renders as a modal. To override this behavior, pass `ADrawer` the `asModal` prop value of `false`. For documentation on modals, reference the [`AModal` documentation](/components/modal).

#### Application Toolbar

<Playground
  code={`function ApplicationToolbar() {
  const { currentTheme } = useATheme();
  const toolbarRef = useRef();
  const [isOpen, setIsOpen] = useState(false);
  const [isSlim, setIsSlim] = useState(true);
  usePopupQuickExit({
    popupRef: toolbarRef,
    isEnabled: isOpen,
    onExit: () => setIsOpen(false),
  });
  const toolbarItems = [
    {
      icon: isSlim ? "caret-right" : "caret-left",
      onClick: () => setIsSlim((state) => !state),
    },
    {
      icon: "envelope",
      text: "Inbox",
    },
    {
      icon: "upload-simple",
      text: "Sent",
    },
    {
      icon: "trash",
      text: "Trash",
    },
    {
      icon: "x",
      onClick: () => setIsOpen(false),
      text: "Close",
    },
  ];
  return (
    <>
      <AButton onClick={() => setIsOpen((prevState) => !prevState)}>
        Open Toolbar
      </AButton>
      <ADrawer
        aria-label="Application toolbar."
        ref={toolbarRef}
        slimWidth="75px"
        openWidth="250px"
        isOpen={isOpen}
        slim={isSlim}
      >
        <AList
          className={
            currentTheme === "default" ? "white" : "mds-neutral--neutral-16"
          }
          style={{ border: "none" }}
        >
          {toolbarItems.map(({ icon, text, onClick, ...rest }) => (
            <AListItem twoLine key={text} {...rest}>
              <AListItemAvatar className="flex-shrink-0 mr-4">
                <AButton
                  onClick={onClick}
                  className={
                    text === "Close"
                      ? "coral--burnt-1 white--text"
                      : "white--text"
                  }
                  icon
                >
                  <AIcon>{icon}</AIcon>
                </AButton>
              </AListItemAvatar>
              {text && <AListItemContent>{text}</AListItemContent>}
            </AListItem>
          ))}
        </AList>
      </ADrawer>
    </>
  );
}
  `}
/>
