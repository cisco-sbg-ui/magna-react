#### Loading States

<Playground
  code={`
function LoadingUpload() {
  const [loading, setLoading] = React.useState(false);

  React.useEffect(() => {
    if (loading) {
      setTimeout(() => {
        setLoading(false);
      }, 5000);
    }
  }, [loading]);

  return (
    <AUpload loading={loading} onUpload={() => setLoading(true)}/>
  );
}
  `}
/>
