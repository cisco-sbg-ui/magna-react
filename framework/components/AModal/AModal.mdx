# Menus

## Install

Integrate with your build to [auto-import](/#integrating), or add an import in your component:

```
import {AModal} from "@cisco-sbg-ui/magna-react";
```

## Usage

<Playground
  fullWidthPreview
  code={`() => {
    const [open, setOpen] = useState(false);
    const ref = useRef();
    useEscapeKeydown({
      isEnabled: open,
      onKeydown: () => setOpen(false)
    })
    useOutsideClick({
      rootRef: ref,
      isEnabled: open,
      onClick: () => setOpen(false)
    })
    return (
      <div>
        <AButton
            onClick={() => setOpen(true)}>
            Open Dialog 1
        </AButton>
        <AModal
            isOpen={open}>
          <DemoModalContent rootRef={ref} onClose={() => setOpen(false)} />
        </AModal>
      </div>
    )
  function DemoModalContent({ rootRef, onClose }) {
    return (
      <APanel ref={rootRef} style={{ minWidth: '400px' }} type="dialog">
        <APanelHeader>
          <APanelTitle>Modal</APanelTitle>
          <AButton onClick={onClose} tertiaryAlt icon>
            <AIcon>close</AIcon>
          </AButton>
        </APanelHeader>
        <APanelBody>
          Modal content goes here.
        </APanelBody>
        <APanelFooter>
          <AButton>Action</AButton>
        </APanelFooter>
      </APanel>
    )
  }
  }
`}
/>

### Form Example

<Playground
  code={`() => {
  const [modalOpen, setModalOpen] = useState(false);
  return (
    <>
      <AButton onClick={() => setModalOpen(true)}>Start</AButton>
      <AModal isOpen={modalOpen}>
        <APageOverlay
            style={{
              display: 'flex',
              alignItems: 'center',
              justifyContent: 'center'}}>
          <APanel style={{width: '700px'}}>
            <APanelHeader>
              <APanelTitle>Embedded Form</APanelTitle>
              <AButton onClick={() => setModalOpen(false)} tertiaryAlt icon>
                <AIcon>close</AIcon>
              </AButton>
            </APanelHeader>
            <APanelBody>
              <FormContent />
            </APanelBody>
          </APanel>
        </APageOverlay>
      </AModal>
    </>
  );
  function FormContent() {
    const form = useRef(null);
    const [email, setEmail] = useState("");
    const [favNum, setFavNum] = useState("");
    const [errors, setErrors] = useState([]);
    const [comments, setComments] = useState("");
    const [role, setRole] = useState(null);
    return (
      <AForm ref={form}>
        <AContainer className="pa-0">
          <ARow>
            <ACol>
              {errors.length > 0 && (
                <AAlert className="my-2" level="danger" dismissable={false}>
                  There are {errors.length} validation errors on the form.
                </AAlert>
              )}
            </ACol>
          </ARow>
          <ARow>
            <ACol cols="12" sm="6">
              <ATextInput
                label="Email"
                required
                maxLength={50}
                value={email}
                onChange={(e) => {
                  setEmail(e.target.value);
                }}
                validateOnBlur
                rules={[
                  {
                    test: (value) =>
                      !["@gmail.com", "@hotmail.com"].some((v) =>
                        value.includes(v)
                      ) || "Please provide a business email address",
                    level: "warning"
                  },
                  {
                    test: (value) =>
                      /^[A-Z0-9._%+-]+@[A-Z0-9.-]+\.[A-Z]{2,}$/i.test(value) ||
                      "Invalid Email",
                    level: "danger"
                  }
                ]}
                hint="Please use a business email address"
              />
            </ACol>
            <ACol>
              <ASelect
                items={[
                  {value: 1, text: "Read"},
                  {value: 2, text: "Write"},
                  {value: 3, text: "Admin"}
                ]}
                label="Role"
                required
                rules={[
                  {
                    test: (v) => v.value !== 3 || "Role is set to Admin",
                    level: "warning"
                  }
                ]}
                onSelected={(item) => setRole(item)}
              />
            </ACol>
          </ARow>
          <ARow>
            <ACol>
              <ATextarea
                label="Comments"
                onChange={(e) => {
                  setComments(e.target.value);
                }}
                value={comments}
                required
                rules={[
                  {
                    test: (v) =>
                      v.length < 10 ||
                      "Comments must be less than 10 characters",
                    level: "danger"
                  }
                ]}
                validateOnBlur
                hint="Keep it short"
              />
            </ACol>
          </ARow>
          <ARow>
            <ACol className="text-right">
              <AButton
                secondary
                className="mr-2"
                onClick={() => {
                  setErrors([]);
                  form.current.reset();
                }}>
                Reset
              </AButton>
              <AButton onClick={() => setErrors(form.current.validate())}>
                Validate
              </AButton>
            </ACol>
          </ARow>
        </AContainer>
      </AForm>
    )
  }
}
`}
/>
